internal fun setMessagesPreservingLocalOnly(
    incoming: List<Message>,
    localOnlyFromDb: List<Message>,
    windowFloor: Date?,
) {
    val incomingIds = incoming.map { it.id }.toSet()
    _messages.update { current ->
        val fromState = current.filter { it.isLocalOnly() }
        val allLocalOnly = (fromState + localOnlyFromDb).distinctBy { it.id }
        val survivingLocalOnly = allLocalOnly.filter { localMsg -> localMsg.id !in incomingIds }
        val inWindowLocalOnly = if (windowFloor == null) survivingLocalOnly
            else survivingLocalOnly.filter { msg ->
                msg.getCreatedAtOrNull()?.let { d -> !d.before(windowFloor) } ?: true
            }
        val validIncoming = incoming.filterNot { shouldIgnoreUpsertion(it) }
        (validIncoming + inWindowLocalOnly).distinctBy { it.id }.sortedWith(MESSAGE_COMPARATOR)
    }
}

    ```kotlin
    /**
     * Updates [_cachedLatestMessages] with [incoming] server messages while preserving any
     * local-only messages currently in the cache or supplied via [localOnlyFromDb].
     *
     * Unlike [upsertCachedLatestMessages], this method does NOT evict local-only messages
     * when the server returns a page without them. Only a COMPLETED server message with the
     * same ID as a local-only cache entry is allowed to replace it (delivery confirmation).
     *
     * No window floor is applied — the cache always represents "latest" messages and local-only
     * messages at any age belong at the latest end (they are user-generated pending/failed actions).
     *
     * Requirements: PRES-06
     */
    internal fun upsertCachedLatestMessagesPreservingLocalOnly(
        incoming: List<Message>,
        localOnlyFromDb: List<Message>,
    ) {
        if (incoming.isEmpty() && localOnlyFromDb.isEmpty()) return
        val incomingIds = incoming.map { it.id }.toSet()
        _cachedLatestMessages.update { current ->
            // Gather local-only from current cache + DB, dedup by id
            val fromCache = current.filter { it.isLocalOnly() }
            val allLocalOnly = (fromCache + localOnlyFromDb).distinctBy { it.id }
            // Server wins on ID collision (delivery confirmation)
            val survivingLocalOnly = allLocalOnly.filter { it.id !in incomingIds }
            // Filter incoming through shouldIgnoreUpsertion (shadow ban, thread-only guard)
            val validIncoming = incoming.filterNot { shouldIgnoreUpsertion(it) }
            // Merge: existing server entries + new incoming page + surviving local-only
            val existingServer = current.filterNot { it.isLocalOnly() }
            (existingServer + validIncoming + survivingLocalOnly)
                .distinctBy { it.id }
                .sortedWith(MESSAGE_COMPARATOR)
                .takeLast(CACHED_LATEST_MESSAGES_LIMIT)
        }
    }
    ```
